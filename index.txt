<body>
    <h1><%= request.getAttribute("pref") %>の市区町村</h1>

    <input type="hidden" id="pref" value="<%= request.getAttribute("pref") %>">

<%
    Map<String, List<String>> municipalityMap =
        (Map<String, List<String>>) request.getAttribute("municipalityMap");
%>

    <!-- 每個行：radio + 行名 + 下拉選單 -->
<%
    for (Map.Entry<String, List<String>> entry : municipalityMap.entrySet()) {
        String row = entry.getKey();
        List<String> municipalities = entry.getValue();
%>
    <div>
        <input type="radio" name="selectedRow" value="<%= row %>" class="row-radio">
        <strong><%= row %></strong>
        <select class="municipality-select" data-row="<%= row %>" disabled>
            <option value="">選択してください</option>
<% 
        for (String m : municipalities) { 
%>
            <option value="<%= m %>"><%= m %></option>
<% 
        } 
%>
        </select>
    </div>
<% 
    } 
%>

    <br>
    <button onclick="goNext()">次へ</button>
    <button onclick="location.href='municipality?pref=<%= request.getAttribute("pref") %>'">戻る</button>

    <p id="errorMsg" style="color:red; display:none;">市区町村を選択してください</p>

    <script>
        // 取得所有的 radio 和 select
        let radios = document.querySelectorAll('.row-radio');
        let selects = document.querySelectorAll('.municipality-select');

        // 當任何 radio 被點選時
        function onRadioChange() {
            let selectedRow = this.value;   // 例如 "あ行"
            // 遍歷每個 select，只啟用屬於該行的
            for (let i = 0; i < selects.length; i++) {
                let select = selects[i];
                if (select.getAttribute('data-row') === selectedRow) {
                    select.disabled = false;
                } else {
                    select.disabled = true;
                    select.value = "";   // 清空其他行的選擇
                }
            }
        }

        // 幫每個 radio 綁定事件
        for (let i = 0; i < radios.length; i++) {
            radios[i].addEventListener('change', onRadioChange);
        }

        // 次へ按鈕處理
        function goNext() {
            // 檢查哪個 radio 被選中
            let selectedRadio = null;
            for (let i = 0; i < radios.length; i++) {
                if (radios[i].checked) {
                    selectedRadio = radios[i];
                    break;
                }
            }

            if (!selectedRadio) {
                document.getElementById('errorMsg').style.display = 'block';
                return;
            }

            // 取得對應的 select 的值
            let selectedRow = selectedRadio.value;
            let targetSelect = null;
            for (let i = 0; i < selects.length; i++) {
                if (selects[i].getAttribute('data-row') === selectedRow) {
                    targetSelect = selects[i];
                    break;
                }
            }

            let selectedValue = targetSelect.value;
            if (selectedValue === "") {
                document.getElementById('errorMsg').style.display = 'block';
                return;
            }

            // 全部正確，跳轉
            document.getElementById('errorMsg').style.display = 'none';
            let pref = document.getElementById('pref').value;
            location.href = 'town?pref=' + encodeURIComponent(pref) + '&municipality=' + encodeURIComponent(selectedValue);
        }
    </script>
</body>
